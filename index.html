Connexion Base de données (Supabase)

Connectez votre propre base de données Supabase pour synchroniser en temps réel les données de la colocation sur tous les appareils.

Synchronisation & Migration

Vous avez des données locales existantes que vous souhaitez exporter vers votre nouvelle base de données Supabase ? Utilisez le bouton ci-dessous.

Initialiser la base de données (SQL)

Copiez le script SQL ci-dessous et exécutez-le dans la console SQL de votre projet Supabase pour créer automatiquement les tables requises.

-- 1. Table des Colocataires
CREATE TABLE IF NOT EXISTS public.cohabit_roommates (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    emoji TEXT,
    hue INTEGER,
    color TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 2. Table du Calendrier (Événements & Absences)
CREATE TABLE IF NOT EXISTS public.cohabit_calendar (
    id TEXT PRIMARY KEY,
    roommate_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE SET NULL,
    type TEXT NOT NULL,
    start_date TEXT NOT NULL,
    start_time TEXT,
    end_date TEXT NOT NULL,
    end_time TEXT,
    note TEXT,
    title TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 3. Table des Tâches Ménagères
CREATE TABLE IF NOT EXISTS public.cohabit_cleaning_tasks (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    description TEXT,
    responsible_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE SET NULL,
    status TEXT NOT NULL,
    last_done TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 4. Table d'Historique du Ménage
CREATE TABLE IF NOT EXISTS public.cohabit_cleaning_history (
    id TEXT PRIMARY KEY,
    task_name TEXT NOT NULL,
    roommate_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE SET NULL,
    date TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 5. Table de Liste de Courses
CREATE TABLE IF NOT EXISTS public.cohabit_shopping (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    category TEXT,
    added_by_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE SET NULL,
    date TEXT NOT NULL,
    checked BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 6. Table des Réparations (Kanban)
CREATE TABLE IF NOT EXISTS public.cohabit_issues (
    id TEXT PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    priority TEXT NOT NULL,
    assignee_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE SET NULL,
    status TEXT NOT NULL,
    date TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 7. Table des Dépenses (Tricount)
CREATE TABLE IF NOT EXISTS public.cohabit_expenses (
    id TEXT PRIMARY KEY,
    title TEXT NOT NULL,
    amount NUMERIC(10, 2) NOT NULL,
    date TEXT NOT NULL,
    payer_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE SET NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 8. Table des Participants aux Dépenses
CREATE TABLE IF NOT EXISTS public.cohabit_expense_participants (
    id TEXT PRIMARY KEY,
    expense_id TEXT REFERENCES public.cohabit_expenses(id) ON DELETE CASCADE,
    roommate_id TEXT REFERENCES public.cohabit_roommates(id) ON DELETE CASCADE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Désactiver la RLS pour simplifier l'accès public
ALTER TABLE public.cohabit_roommates DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_calendar DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_cleaning_tasks DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_cleaning_history DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_shopping DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_issues DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_expenses DISABLE ROW LEVEL SECURITY;
ALTER TABLE public.cohabit_expense_participants DISABLE ROW LEVEL SECURITY;
